pattern_CN = re.compile(r"[\u4e00-\u9fa5]")
pattern_init = re.compile(r"^[n|v|a]")

def tokenize(text):
    kept_words = list()
    for word, flag in pseg.cut(OpenCC("t2s").convert(str(text).strip())):
        if (word not in stopwords) and \
        (len(word) > 1) and \
        (pattern_CN.search(word) != None) and \
        (pattern_init.search(flag) != None):
            kept_words.append(word)
    return " ".join(kept_words)

  • topic coherence

    • Semantic coherence is a criterion developed by Mimno et al. (2011) and is closely related to PMI (pointwise mutual information), it is maximized when the most probable words in a given topic frequently co-occur together. Furthermore, Mimno et al. (2011) showed that this metric correlates well with human judgment of topic quality.

  • perplexity

      • Perplexity is not strongly correlated to human judgment.

import gensim

# Gensim API doc
def evaluate(corpus, dictionary, k, a, b):
    # LDA
    model = gensim.models.LdaMulticore(corpus=corpus, # doc
                                       id2word=dictionary, # 
                                       num_topics=k, # 
                                       alpha=a, # doc-topic
                                       eta=b, # topic-word
                                       random_state=110)
    perplexity = model.log_perplexity(corpus)
    coherence = gensim.models.CoherenceModel(model=model, # 
                                             texts=corpus_raw, # 
                                             dictionary=dictionary, # 
                                             coherence="c_v") # 
    return coherence.get_coherence(), perplexity

def summ(ldamodel, corpus, texts): # corpus
    doc_topic = pd.DataFrame()
    for i, result in enumerate(ldamodel[corpus]):
        field = result[0] # 
        field = sorted(field, key=lambda x: x[1], reverse=True) # 
        for idx, (topic_id, topic_prob) in enumerate(field): # 
            if idx == 0: # 
                word_prob = ldamodel.show_topic(topic_id) # show_topic
                topic_words = ", ".join([word for word, prob in word_prob])
                # doc123
                doc_topic = doc_topic.append(pd.Series([int(topic_id), 
                                                        round(topic_prob,4), 
                                                        topic_words]), 
                                                       ignore_index=True)
            else:
                break
    doc_topic.columns = ['topic_dominant', 'topic_prob', 'topic_words']
    contents = pd.Series(texts) # texts
    doc_topic = pd.concat([doc_topic, contents], axis=1) # 
    return(doc_topic)

  •  perplexity
      

    • A common quantitative way to evaluate topic models is to split each document into a training set and a test set, learn a topic model on the training portion of each document, and compute the probability of the held out word counts under the model. A slight variation of this probability is called "perplexity". Lower values are better. Estimates of this quantity are provided during training.
