  • 下表根据这份 参考资料 绘制
特征 有监督 无监督
标签 有标签的数据用于训练，生成最优模型（类似“热启动”） 没有打标签和训练过程，直接在数据上建模（类似”冷启动“）

1.1 通过 Gensim 基础来理解文本的向量化表示¶

  • 明确一些基本概念

    • 文档（document）：文字集合（如：一条推文）

    • 语料（corpus）：一系列文档构成的集合

    • 向量（vector）：对于文档的数学／数字化表示

    • 模型（model）：一种向量转化算法，可以将某文档对应的向量转化为其他类型的向量

  • 以下示例语料来自 清华新闻网

# 引入需要的包
import jieba

# 简写一个双循环
# 第一层循环：循环doc，对应for doc in corpus
# 第二层循环：循环doc的分词结果，对应for word in jieba.cut(doc)
texts = [[word for word in jieba.cut(doc)] for doc in corpus]

# 词频统计
from collections import defaultdict
# 这是和第一讲一样的操作方式
freq = defaultdict(int)
# 循环展开，思考简写模式和展开模式的写法异同
for text in texts:
    for token in text:
        freq[token] += 1

# 双循环 + if判断，结合了上一步的词频统计结果
processed_corpus = [[word for word in text if len(word) > 1 and freq[word] > 1] for text in texts]

for item in processed_corpus:
    print(item)

# 我们将从基础的词袋表示、TF-IDF表示过渡到更高级的语义级别表示（i.e., "in a more semantic way"）
# 启动一个模型／转换器, LSI = latent semantic indexing（潜在语义索引）
# 此处的操作和上面的BoW转化为TF-IDF是一样的
lsi_model = models.LsiModel(tfidf[bow_corpus], # 传入语料的向量表示
                            id2word=dict_, # 传入词典，一般在降维过程中，需要传入的参数都会比普通操作多一些
                            num_topics=2) # 传入主题数量，主题数量属于超参数，需要人为指定，主题建模本身就是在降维，我们希望得到两条语义主轴

pattern_CN = re.compile(r"[\u4e00-\u9fa5]")
pattern_init = re.compile(r"^[n|v|a]")

def tokenize(text):
    kept_words = list()
    for word, flag in pseg.cut(OpenCC("t2s").convert(str(text).strip())):
        if (word not in stopwords) and \
        (len(word) > 1) and \
        (pattern_CN.search(word) != None) and \
        (pattern_init.search(flag) != None):
            kept_words.append(word)
    return " ".join(kept_words)

  • 主题连贯性（topic coherence）：用于衡量主题模型的质量，表示一个主题下高概率词语间的语义相似性

    • 有 不同的衡量方式
    • 权威解释：Semantic coherence is a criterion developed by Mimno et al. (2011) and is closely related to PMI (pointwise mutual information), it is maximized when the most probable words in a given topic frequently co-occur together. Furthermore, Mimno et al. (2011) showed that this metric correlates well with human judgment of topic quality.

  • 另一个衡量主题模型质量的指标：困惑度（perplexity）

      • 尤其需要注意：Perplexity is not strongly correlated to human judgment.

# 构建一个函数用来计算主题连贯性、困惑度两项指标
import gensim

# 关于具体的参数含义，请阅读Gensim API doc
def evaluate(corpus, dictionary, k, a, b):
    # 使用多核LDA主题建模
    model = gensim.models.LdaMulticore(corpus=corpus, # doc的向量表示
                                       id2word=dictionary, # 词典
                                       num_topics=k, # 主题数量
                                       alpha=a, # 掌管doc-topic分布的参数
                                       eta=b, # 掌管topic-word分布的参数
                                       random_state=110)
    perplexity = model.log_perplexity(corpus)
    coherence = gensim.models.CoherenceModel(model=model, # 传入建构好的模型
                                             texts=corpus_raw, # 分词后的语料
                                             dictionary=dictionary, # 词典
                                             coherence="c_v") # 主题连贯性计算方式
    return coherence.get_coherence(), perplexity

# tqdm：进度条，适合需要长时间运算的模型
import tqdm

# 需要调整的参数汇总，范围需要自定义——“网格搜索”

beta = ["symmetric", "auto"]

# 存储运行结果
results = {"topics": list(), 
           "alpha": list(), 
           "beta": list(), 
           "coherence": list(), 
           "perplexity": list()}

# 设置进度条长度
pbar = tqdm.tqdm(total=8*2*2)
# 三层嵌套式for循环
for k in topics_range:
    for a in alpha:
        for b in beta:
            cp = evaluate(corpus=corpus_bow, 
                          dictionary=dict_, 
                          k=k, 
                          a=a, 
                          b=b)
            results["topics"].append(k)
            results["alpha"].append(a)
            results["beta"].append(b)
            results["coherence"].append(cp[0])
            results["perplexity"].append(cp[1])
            # 一个组合过后，进度条推进一格
            pbar.update(1)
pbar.close()

# 综合以上结果，整理一张表格，展示每一则doc对应的最显著主题及主题下的高概率词
def summ(ldamodel, corpus, texts): # 形参：模型、向量化后的语料、原始corpus
    doc_topic = pd.DataFrame()
    for i, result in enumerate(ldamodel[corpus]):
        field = result[0] # 获取输出结果中的第一个要素，也即由（主题编号，主题概率）构成的列表
        field = sorted(field, key=lambda x: x[1], reverse=True) # 按照主题概率从大到小进行排序
        for idx, (topic_id, topic_prob) in enumerate(field): # 对排序好的结果进行循环
            if idx == 0: # 针对最显著的主题进行操作，这是最显著的主题已经排在第一位了
                word_prob = ldamodel.show_topic(topic_id) # 此处的show_topic反映的是主题下的（词语，词语概率）
                topic_words = ", ".join([word for word, prob in word_prob])
                # 在数据框中追加每一则doc对应的：1）最显著的主题编号；2）最显著的主题的概率；3）该主题对应的高概率词
                doc_topic = doc_topic.append(pd.Series([int(topic_id), 
                                                        round(topic_prob,4), 
                                                        topic_words]), 
                                                       ignore_index=True)
            else:
                break
    doc_topic.columns = ['topic_dominant', 'topic_prob', 'topic_words']
    contents = pd.Series(texts) # 这里的texts对应的是未经任何处理的原始语料
    doc_topic = pd.concat([doc_topic, contents], axis=1) # 执行合并
    return(doc_topic)


df_summ = summ(ldamodel=final_model, corpus=corpus_bow, texts=meta["content"].to_list()) # 调用定义的函数

import numpy as np

from bokeh.io import push_notebook, show, output_notebook

output_notebook()

# 获取每一则doc对应最显著主题的概率
topic_prob = list()
for i, result in enumerate(final_model[corpus_bow]):
    topic_prob.append([weight for i_, weight in result[0]])

# 生成一个数组，每一行为每一则doc在每一主题下的概率，维数 = 1985*4
topic_prob_array = pd.DataFrame(topic_prob).fillna(0).values
# print(topic_prob_array, len(topic_prob_array))

# 为了绘图的美观，可以人为设定一个阈值，检验每一则doc对应概率中的最大值是否大于阈值——大于：保留、小于：舍弃
topic_prob_array = topic_prob_array[np.amax(topic_prob_array, axis=1) > 0.30]
# print(topic_prob_array, len(topic_prob_array))

# 获取每一则doc中最显要的主题编号，实际上就是最大值对应的索引
topic_id = np.argmax(topic_prob_array, axis=1)

# 采用tSNE进行降维，把docs压缩到二维平面上, tSNE = t-distributed Stochastic Neighbor Embedding.
tsne_model = TSNE(n_components=2, # 指定压缩至二维平面
                  random_state=120, 
                  init='pca', # 指定降维方法
                  n_jobs=-1, # 加快运行效率，使用所有处理器
                  perplexity=50) # 影响邻居的数量，越大的数据需要越高的困惑度数值（和Gensim中的perplexity不是一回事）
# 将设定好的降维模型应用在数组上，原始的概率数组会被压缩为二维数据
tsne_tm = tsne_model.fit_transform(topic_prob_array)
# print(tsne_tm, len(tsne_tm))

topic_num = 4
# 获取一系列matplotlib中色板的颜色，共有10种颜色
mycolors = np.array([color for name, color in mcolors.TABLEAU_COLORS.items()])
plot = figure(title="t-SNE visualization of topic num. = %s"%topic_num,
              toolbar_location="above",
              plot_width=600, 
              plot_height=600)
plot.scatter(x=tsne_tm[:,0], # 把（X，Y）投射到坐标空间
             y=tsne_tm[:,1], 
             color=mycolors[topic_id])
show(plot)
